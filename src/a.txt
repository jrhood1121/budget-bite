import { useMemo, useState } from 'react'

type GroceryItem = {
  name: string
  price: number
  servings: number
  category: string
  quantity: number
}

type GroceryResponse = {
  budget: number
  mealsPerWeek: number
  totalEstimatedCost: number
  estimatedCostPerMeal: string
  totalEstimatedServings: number
  groceryList: GroceryItem[]
  message: string
}

export default function App() {
  const [budget, setBudget] = useState(100)
  const [mealsPerWeek, setMealsPerWeek] = useState(21)

  const [vegetarian, setVegetarian] = useState(false)
  const [vegan, setVegan] = useState(false)
  const [glutenFree, setGlutenFree] = useState(false)
  const [dairyFree, setDairyFree] = useState(false)

  const [halal, setHalal] = useState(false)
  const [kosher, setKosher] = useState(false)

  const [avoidFoods, setAvoidFoods] = useState('')
  const [nutritionGoal, setNutritionGoal] = useState('')
  const [itemsInFridge, setItemsInFridge] = useState('')

  const [results, setResults] = useState<GroceryResponse | null>(null)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState('')

  const costPerMeal =
    mealsPerWeek > 0 ? (budget / mealsPerWeek).toFixed(2) : '0.00'

  const remainingBudget = results
    ? (results.budget - results.totalEstimatedCost).toFixed(2)
    : '0.00'

  const selectedPreferenceLabel = useMemo(() => {
    const labels: string[] = []

    if (vegan) labels.push('vegan')
    else if (vegetarian) labels.push('vegetarian')

    if (glutenFree) labels.push('gluten-free')
    if (dairyFree) labels.push('dairy-free')
    if (halal) labels.push('halal')
    if (kosher) labels.push('kosher')

    return labels.length > 0 ? labels.join(', ') : 'your preferences'
  }, [vegan, vegetarian, glutenFree, dairyFree, halal, kosher])

  const groupedItems = useMemo(() => {
    if (!results) return {}

    const categoryLabels: Record<string, string> = {
      protein: 'Protein',
      carb: 'Carbs',
      fruit: 'Fruit',
      vegetable: 'Vegetables',
      dairy: 'Dairy',
      other: 'Other',
    }

    return results.groceryList.reduce<Record<string, GroceryItem[]>>(
      (acc, item) => {
        const label = categoryLabels[item.category] ?? 'Other'
        if (!acc[label]) acc[label] = []
        acc[label].push(item)
        return acc
      },
      {}
    )
  }, [results])

  function getQuantityLabel(item: GroceryItem) {
    if (item.name === 'Eggs') {
      return `${item.quantity} carton${item.quantity > 1 ? 's' : ''}`
    }

    if (item.name === 'Milk') {
      return `${item.quantity} gallon${item.quantity > 1 ? 's' : ''}`
    }

    if (
      item.name === 'Beans' ||
      item.name === 'Black Beans' ||
      item.name === 'Chickpeas' ||
      item.name === 'Tuna'
    ) {
      return `${item.quantity} can${item.quantity > 1 ? 's' : ''}`
    }

    if (
      item.name === 'Greek Yogurt' ||
      item.name === 'Yogurt' ||
      item.name === 'Cottage Cheese'
    ) {
      return `${item.quantity} tub${item.quantity > 1 ? 's' : ''}`
    }

    if (
      item.name === 'Bread' ||
      item.name === 'Whole Grain Bread'
    ) {
      return `${item.quantity} loaf${item.quantity > 1 ? 'ves' : ''}`
    }

    if (
      item.name === 'Tofu' ||
      item.name === 'Tempeh'
    ) {
      return `${item.quantity} block${item.quantity > 1 ? 's' : ''}`
    }

    if (
      item.name === 'Rice' ||
      item.name === 'Brown Rice' ||
      item.name === 'Quinoa' ||
      item.name === 'Lentils' ||
      item.name === 'Oats'
    ) {
      return `${item.quantity} bag${item.quantity > 1 ? 's' : ''}`
    }

    return `${item.quantity} item${item.quantity > 1 ? 's' : ''}`
  }

  function getLineTotal(item: GroceryItem) {
    return (item.quantity * item.price).toFixed(2)
  }

  async function handleGenerateList() {
    setLoading(true)
    setError('')
    setResults(null)

    try {
      const response = await fetch('http://localhost:3001/api/grocery-list', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          budget,
          mealsPerWeek,
          dietaryPreferences: {
            vegetarian,
            vegan,
            glutenFree,
            dairyFree,
          },
          religiousPreferences: {
            halal,
            kosher,
          },
          avoidFoods,
          nutritionGoal,
          itemsInFridge,
        }),
      })

      if (!response.ok) {
        throw new Error('Failed to generate grocery list')
      }

      const data: GroceryResponse = await response.json()
      setResults(data)
    } catch (err) {
      setError('Something went wrong while getting your grocery list.')
      console.error(err)
    } finally {
      setLoading(false)
    }
  }

  function handleReset() {
    setBudget(100)
    setMealsPerWeek(21)

    setVegetarian(false)
    setVegan(false)
    setGlutenFree(false)
    setDairyFree(false)

    setHalal(false)
    setKosher(false)

    setAvoidFoods('')
    setNutritionGoal('')
    setItemsInFridge('')

    setResults(null)
    setError('')
    setLoading(false)
  }

  return (
    <div className="page">
      <div className="container">
        <header className="hero">
          <div className="brandRow">
            <span className="brandIcon">🧺</span>
            <h1 className="title">Budget Bite</h1>
          </div>

          <p className="subtitle">
            Plan your weekly groceries based on your budget and dietary needs
          </p>
        </header>

        <div className="card">
          <h2 className="cardTitle">Grocery List Generator</h2>
          <p className="cardSubtitle">
            Enter your budget and preferences to get a customized shopping list
          </p>

          <div className="topInputBox">
            <div className="inputRow">
              <div className="field">
                <label htmlFor="budget">Weekly Budget ($) *</label>
                <input
                  id="budget"
                  type="number"
                  value={budget}
                  onChange={(e) => setBudget(Number(e.target.value))}
                />
              </div>

              <div className="field">
                <label htmlFor="meals">Meals per Week *</label>
                <input
                  id="meals"
                  type="number"
                  value={mealsPerWeek}
                  onChange={(e) => setMealsPerWeek(Number(e.target.value))}
                />
                <p className="fieldHelp">
                  For family planning: multiply people × meals/day × 7
                </p>
              </div>
            </div>
          </div>

          <details className="sectionDropdown">
            <summary className="sectionSummary">
              <span>Dietary Preferences (Optional)</span>
              <span className="chevron">⌄</span>
            </summary>
            <div className="sectionContent">
              <label className="checkboxRow">
                <input
                  type="checkbox"
                  checked={vegetarian}
                  onChange={(e) => setVegetarian(e.target.checked)}
                />
                Vegetarian
              </label>
              <label className="checkboxRow">
                <input
                  type="checkbox"
                  checked={vegan}
                  onChange={(e) => setVegan(e.target.checked)}
                />
                Vegan
              </label>
              <label className="checkboxRow">
                <input
                  type="checkbox"
                  checked={glutenFree}
                  onChange={(e) => setGlutenFree(e.target.checked)}
                />
                Gluten-Free
              </label>
              <label className="checkboxRow">
                <input
                  type="checkbox"
                  checked={dairyFree}
                  onChange={(e) => setDairyFree(e.target.checked)}
                />
                Dairy-Free
              </label>
            </div>
          </details>

          <details className="sectionDropdown">
            <summary className="sectionSummary">
              <span>Religious Dietary Preferences (Optional)</span>
              <span className="chevron">⌄</span>
            </summary>
            <div className="sectionContent">
              <label className="checkboxRow">
                <input
                  type="checkbox"
                  checked={halal}
                  onChange={(e) => setHalal(e.target.checked)}
                />
                Halal
              </label>
              <label className="checkboxRow">
                <input
                  type="checkbox"
                  checked={kosher}
                  onChange={(e) => setKosher(e.target.checked)}
                />
                Kosher
              </label>
            </div>
          </details>

          <details className="sectionDropdown">
            <summary className="sectionSummary">
              <span>Allergies / Foods to Avoid (Optional)</span>
              <span className="chevron">⌄</span>
            </summary>
            <div className="sectionContent">
              <textarea
                className="sectionTextarea"
                placeholder="Example: peanuts, shellfish, mushrooms"
                value={avoidFoods}
                onChange={(e) => setAvoidFoods(e.target.value)}
              />
            </div>
          </details>

          <details className="sectionDropdown">
            <summary className="sectionSummary">
              <span>Nutritional Goals (Optional)</span>
              <span className="chevron">⌄</span>
            </summary>
            <div className="sectionContent">
              <textarea
                className="sectionTextarea"
                placeholder="Example: high protein, low carb"
                value={nutritionGoal}
                onChange={(e) => setNutritionGoal(e.target.value)}
              />
            </div>
          </details>

          <details className="sectionDropdown">
            <summary className="sectionSummary">
              <span>Items Already in Fridge (Optional)</span>
              <span className="chevron">⌄</span>
            </summary>
            <div className="sectionContent">
              <textarea
                className="sectionTextarea"
                placeholder="Example: eggs, milk, rice"
                value={itemsInFridge}
                onChange={(e) => setItemsInFridge(e.target.value)}
              />
            </div>
          </details>

          <div className="buttonRow">
            <button className="generateButton" onClick={handleGenerateList}>
              Generate Grocery List
            </button>

            <button className="resetButton" onClick={handleReset} type="button">
              Reset
            </button>
          </div>

          <p className="costText">Estimated budget per meal: ${costPerMeal}</p>

          {loading && <p className="resultsText">Loading...</p>}
          {error && <p className="resultsText errorText">{error}</p>}

          {results && (
            <div className="resultsPanel">
              <div className="resultsHeader">
                <h3 className="resultsHeading">🛒 Your Grocery List</h3>
                <p className="tailoredText">
                  Tailored for: <span>{selectedPreferenceLabel}</span>
                </p>
              </div>

              <div className="summaryGrid">
                <div className="summaryCard">
                  <div className="summaryLabel">$ Budget</div>
                  <div className="summaryValue">${results.budget.toFixed(2)}</div>
                </div>

                <div className="summaryCard">
                  <div className="summaryLabel">📦 Total Cost</div>
                  <div className="summaryValue">
                    ${results.totalEstimatedCost.toFixed(2)}
                  </div>
                </div>

                <div className="summaryCard">
                  <div className="summaryLabel">Remaining</div>
                  <div className="summaryValue summaryGreen">
                    ${remainingBudget}
                  </div>
                </div>
              </div>

              <div className="goalsSection">
                <h4 className="goalsTitle">Your Nutritional Goals</h4>

                <div className="goalsGrid">
                  <div className="goalCard">
                    <div className="goalLabel">🍽 Meals</div>
                    <div className="goalValue">{results.mealsPerWeek}/wk</div>
                  </div>

                  <div className="goalCard">
                    <div className="goalLabel">Calories</div>
                    <div className="goalValue">2000/day</div>
                  </div>

                  <div className="goalCard">
                    <div className="goalLabel">Protein</div>
                    <div className="goalValue">
                      {nutritionGoal.toLowerCase().includes('high protein')
                        ? '90g'
                        : '50g'}
                    </div>
                  </div>

                  <div className="goalCard">
                    <div className="goalLabel">Carbs</div>
                    <div className="goalValue">250g</div>
                  </div>

                  <div className="goalCard">
                    <div className="goalLabel">Fat</div>
                    <div className="goalValue">65g</div>
                  </div>
                </div>
              </div>

              {Object.entries(groupedItems).map(([category, items]) => (
                <div key={category} className="categorySection">
                  <div className="categoryHeader">
                    <span className="categoryPill">{category}</span>
                    <span className="categoryCount">
                      {items.length} item{items.length !== 1 ? 's' : ''}
                    </span>
                  </div>

                  <div className="itemList">
                    {items.map((item) => (
                      <div key={item.name} className="itemCard">
                        <div>
                          <div className="itemName">{item.name}</div>
                          <div className="itemMeta">{getQuantityLabel(item)}</div>
                        </div>

                        <div className="itemRight">
                          <div className="itemPrice">${getLineTotal(item)}</div>
                          <button className="swapButton" type="button">
                            ↻ Swap
                          </button>
                        </div>
                      </div>
                    ))}
                  </div>
                </div>
              ))}
            </div>
          )}
        </div>
      </div>
    </div>
  )
}
